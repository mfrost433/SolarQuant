SELECT cip.node_id, cip.start_datetime, cip.end_datetime, (cip.kilowatt_hours_weight*40) as kilowatt_hours FROM consumption_input_pattern cip

SELECT cip.node_id, cip.start_datetime, cip.end_datetime, (cip.kilowatt_hours_weight*40) as kilowatt_hours FROM consumption_input_pattern cip 
				INNER JOIN inputpattern_extensions ie
				ON cip.consumption_input_pattern_id = ie.consumption_input_pattern_id				
				where cip.node_id = 205 AND cip.start_datetime > '2017-01-01' and cip.start_datetime < '2017-12-31' ORDER BY cip.start_datetime ASC

SELECT cip.node_id, cip.start_datetime, cip.end_datetime, (cip.kilowatt_hours_weight*40) as kilowatt_hours, pnm.sourceId FROM consumption_input_pattern cip 
				INNER JOIN inputpattern_extensions ie
				ON cip.consumption_input_pattern_id = ie.consumption_input_pattern_id
				INNER JOIN patternset_node_match pnm
				ON cip.pattern_set_id = pnm.pattern_set_id			
				where cip.node_id = 205 AND cip.start_datetime > '2017-01-01' and cip.start_datetime < '2017-12-31' ORDER BY cip.start_datetime ASC
